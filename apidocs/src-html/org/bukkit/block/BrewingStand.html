001package org.bukkit.block;
002
003import org.bukkit.inventory.BrewerInventory;
004
005/**
006 * Represents a brewing stand.
007 */
008public interface BrewingStand extends BlockState, ContainerBlock {
009
010    /**
011     * How much time is left in the brewing cycle
012     *
013     * @return Brew Time
014     */
015    int getBrewingTime();
016
017    /**
018     * Set the time left before brewing completes.
019     *
020     * @param brewTime Brewing time
021     */
022    void setBrewingTime(int brewTime);
023
024    public BrewerInventory getInventory();
025}
