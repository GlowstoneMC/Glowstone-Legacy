003/**
004 * StringPrompt is the base class for any prompt that accepts an arbitrary
005 * string from the user.
006 */
007public abstract class StringPrompt implements Prompt{
008
009    /**
010     * Ensures that the prompt waits for the user to provide input.
011     *
012     * @param context Context information about the conversation.
013     * @return True.
014     */
015    public boolean blocksForInput(ConversationContext context) {
016        return true;
017    }
018}
