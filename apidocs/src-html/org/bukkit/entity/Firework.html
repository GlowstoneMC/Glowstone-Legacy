001package org.bukkit.entity;
002
003import org.bukkit.inventory.meta.FireworkMeta;
004
005public interface Firework extends Entity {
006
007    /**
008     * Get a copy of the fireworks meta
009     *
010     * @return A copy of the current Firework meta
011     */
012    FireworkMeta getFireworkMeta();
013
014    /**
015     * Apply the provided meta to the fireworks
016     *
017     * @param meta The FireworkMeta to apply
018     */
019    void setFireworkMeta(FireworkMeta meta);
020
021    /**
022     * Cause this firework to explode at earliest opportunity, as if it has no
023     * remaining fuse.
024     */
025    void detonate();
026}
