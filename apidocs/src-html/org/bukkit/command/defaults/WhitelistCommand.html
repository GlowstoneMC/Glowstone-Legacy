001package org.bukkit.command.defaults;
002
003import java.util.ArrayList;
004import java.util.List;
005
006import org.apache.commons.lang.Validate;
007import org.bukkit.Bukkit;
008import org.bukkit.ChatColor;
009import org.bukkit.OfflinePlayer;
010import org.bukkit.command.Command;
011import org.bukkit.command.CommandSender;
012import org.bukkit.util.StringUtil;
013
014import com.google.common.collect.ImmutableList;
015
016public class WhitelistCommand extends VanillaCommand {
017    private static final List<String> WHITELIST_SUBCOMMANDS = ImmutableList.of("add", "remove", "on", "off", "list", "reload");
018
019    public WhitelistCommand() {
020        super("whitelist");
021        this.description = "Manages the list of players allowed to use this server";
022        this.usageMessage = "/whitelist (add|remove) <player>\n/whitelist (on|off|list|reload)";
023        this.setPermission("bukkit.command.whitelist.reload;bukkit.command.whitelist.enable;bukkit.command.whitelist.disable;bukkit.command.whitelist.list;bukkit.command.whitelist.add;bukkit.command.whitelist.remove");
024    }
025
026    @Override
027    public boolean execute(CommandSender sender, String currentAlias, String[] args) {
028        if (!testPermission(sender)) return true;
029
030        if (args.length == 1) {
031            if (args[0].equalsIgnoreCase("reload")) {
032                if (badPerm(sender, "reload")) return true;
033
034                Bukkit.reloadWhitelist();
035                Command.broadcastCommandMessage(sender, "Reloaded white-list from file");
036                return true;
037            } else if (args[0].equalsIgnoreCase("on")) {
038                if (badPerm(sender, "enable")) return true;
039
040                Bukkit.setWhitelist(true);
041                Command.broadcastCommandMessage(sender, "Turned on white-listing");
042                return true;
043            } else if (args[0].equalsIgnoreCase("off")) {
044                if (badPerm(sender, "disable")) return true;
045
046                Bukkit.setWhitelist(false);
047                Command.broadcastCommandMessage(sender, "Turned off white-listing");
048                return true;
049            } else if (args[0].equalsIgnoreCase("list")) {
050                if (badPerm(sender, "list")) return true;
051
052                if (Bukkit.getWhitelistedPlayers().size() == 0) {
053                    sender.sendMessage("No players have been white-listed");
054                } else {
055                    StringBuilder result = new StringBuilder();
056
057                    for (OfflinePlayer player : Bukkit.getWhitelistedPlayers()) {
058                        if (result.length() > 0) {
059                            result.append(", ");
060                        }
061
062                        result.append(player.getName());
063                    }
064
065                    sender.sendMessage("White-listed players (" + Bukkit.getWhitelistedPlayers().size() + "): " + result.toString());
066                }
067
068                return true;
069            }
070        } else if (args.length == 2) {
071            if (args[0].equalsIgnoreCase("add")) {
072                if (badPerm(sender, "add")) return true;
073
074                OfflinePlayer target = Bukkit.getOfflinePlayer(args[1]);
075
076                if (target.isWhitelisted()) {
077                    sender.sendMessage(target.getName() + " is already on the white-list");
078                } else {
079                    target.setWhitelisted(true);
080
081                    Command.broadcastCommandMessage(sender, "Added " + args[1] + " to the white-list");
082                }
083
084                return true;
085            } else if (args[0].equalsIgnoreCase("remove")) {
086                if (badPerm(sender, "remove")) return true;
087
088                OfflinePlayer target = Bukkit.getOfflinePlayer(args[1]);
089
090                if (target.isWhitelisted()) {
091                    target.setWhitelisted(false);
092
093                    Command.broadcastCommandMessage(sender, "Removed " + target.getName() + " from the white-list");
094                } else {
095                    sender.sendMessage(target.getName() + " is not on the white-list");
096                }
097                return true;
098            }
099        }
100
101        sender.sendMessage(ChatColor.RED + "Correct command usage:\n" + usageMessage);
102        return false;
103    }
104
105    private boolean badPerm(CommandSender sender, String perm) {
106        if (!sender.hasPermission("bukkit.command.whitelist." + perm)) {
107            sender.sendMessage(ChatColor.RED + "You do not have permission to perform this action.");
108            return true;
109        }
110
111        return false;
112    }
113
114    @Override
115    public List<String> tabComplete(CommandSender sender, String alias, String[] args) {
116        Validate.notNull(sender, "Sender cannot be null");
117        Validate.notNull(args, "Arguments cannot be null");
118        Validate.notNull(alias, "Alias cannot be null");
119
120        if (args.length == 1) {
121            return StringUtil.copyPartialMatches(args[0], WHITELIST_SUBCOMMANDS, new ArrayList<String>(WHITELIST_SUBCOMMANDS.size()));
122        } else if (args.length == 2) {
123            if (args[0].equalsIgnoreCase("add")) {
124                List<String> completions = new ArrayList<String>();
125                for (OfflinePlayer player : Bukkit.getOfflinePlayers()) {
126                    String name = player.getName();
127                    if (StringUtil.startsWithIgnoreCase(name, args[1]) && !player.isWhitelisted()) {
128                        completions.add(name);
129                    }
130                }
131                return completions;
132            } else if (args[0].equalsIgnoreCase("remove")) {
133                List<String> completions = new ArrayList<String>();
134                for (OfflinePlayer player : Bukkit.getWhitelistedPlayers()) {
135                    String name = player.getName();
136                    if (StringUtil.startsWithIgnoreCase(name, args[1])) {
137                        completions.add(name);
138                    }
139                }
140                return completions;
141            }
142        }
143        return ImmutableList.of();
144    }
145}
