001package org.bukkit.help;
002
003import org.apache.commons.lang.StringUtils;
004import org.bukkit.ChatColor;
005import org.bukkit.command.Command;
006import org.bukkit.command.CommandSender;
007import org.bukkit.command.ConsoleCommandSender;
008import org.bukkit.command.defaults.VanillaCommand;
009
010/**
011 * Lacking an alternative, the help system will create instances of
012 * GenericCommandHelpTopic for each command in the server's CommandMap. You
013 * can use this class as a base class for custom help topics, or as an example
014 * for how to write your own.
015 */
016public class GenericCommandHelpTopic extends HelpTopic {
017
018    protected Command command;
019
020    public GenericCommandHelpTopic(Command command) {
021        this.command = command;
022
023        if (command.getLabel().startsWith("/")) {
024            name = command.getLabel();
025        } else {
026            name = "/" + command.getLabel();
027        }
028
029        // The short text is the first line of the description
030        int i = command.getDescription().indexOf("\n");
031        if (i > 1) {
032            shortText = command.getDescription().substring(0, i - 1);
033        } else {
034            shortText = command.getDescription();
035        }
036
037        // Build full text
038        StringBuffer sb = new StringBuffer();
039
040        sb.append(ChatColor.GOLD);
041        sb.append("Description: ");
042        sb.append(ChatColor.WHITE);
043
044        if (command.getDescription().isEmpty()) {
045            sb.append(ChatColor.ITALIC);
046            sb.append("No information available.");
047        } else {
048            sb.append(command.getDescription());
049        }
050
051        sb.append("\n");
052
053        sb.append(ChatColor.GOLD);
054        sb.append("Usage: ");
055        sb.append(ChatColor.WHITE);
056
057        if (command.getUsage().isEmpty()) {
058            sb.append(ChatColor.ITALIC);
059            sb.append("No information available.");
060        } else {
061            sb.append(command.getUsage().replace("<command>", name.substring(1)));
062        }
063
064        if (command.getAliases().size() > 0) {
065            sb.append("\n");
066            sb.append(ChatColor.GOLD);
067            sb.append("Aliases: ");
068            sb.append(ChatColor.WHITE);
069            sb.append(ChatColor.WHITE + StringUtils.join(command.getAliases(), ", "));
070        }
071        fullText = sb.toString();
072    }
073
074    public boolean canSee(CommandSender sender) {
075        if (!command.isRegistered() && !(command instanceof VanillaCommand)) {
076            // Unregistered commands should not show up in the help (ignore VanillaCommands)
077            return false;
078        }
079
080        if (sender instanceof ConsoleCommandSender) {
081            return true;
082        }
083
084        if (amendedPermission != null) {
085            return sender.hasPermission(amendedPermission);
086        } else {
087            return command.testPermissionSilent(sender);
088        }
089    }
090}
