001package org.bukkit.configuration;
002
003/**
004 * Various settings for controlling the input and output of a {@link
005 * MemoryConfiguration}
006 */
007public class MemoryConfigurationOptions extends ConfigurationOptions {
008    protected MemoryConfigurationOptions(MemoryConfiguration configuration) {
009        super(configuration);
010    }
011
012    @Override
013    public MemoryConfiguration configuration() {
014        return (MemoryConfiguration) super.configuration();
015    }
016
017    @Override
018    public MemoryConfigurationOptions copyDefaults(boolean value) {
019        super.copyDefaults(value);
020        return this;
021    }
022
023    @Override
024    public MemoryConfigurationOptions pathSeparator(char value) {
025        super.pathSeparator(value);
026        return this;
027    }
028}
