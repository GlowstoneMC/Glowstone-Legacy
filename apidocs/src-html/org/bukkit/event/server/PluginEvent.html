001package org.bukkit.event.server;
002
003import org.bukkit.plugin.Plugin;
004
005/**
006 * Used for plugin enable and disable events
007 */
008public abstract class PluginEvent extends ServerEvent {
009    private final Plugin plugin;
010
011    public PluginEvent(final Plugin plugin) {
012        this.plugin = plugin;
013    }
014
015    /**
016     * Gets the plugin involved in this event
017     *
018     * @return Plugin for this event
019     */
020    public Plugin getPlugin() {
021        return plugin;
022    }
023}
