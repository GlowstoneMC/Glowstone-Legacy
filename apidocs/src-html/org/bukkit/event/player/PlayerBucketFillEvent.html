001package org.bukkit.event.player;
002
003import org.bukkit.Material;
004import org.bukkit.block.Block;
005import org.bukkit.block.BlockFace;
006import org.bukkit.entity.Player;
007import org.bukkit.event.HandlerList;
008import org.bukkit.inventory.ItemStack;
009
010/**
011 * Called when a player fills a bucket
012 */
013public class PlayerBucketFillEvent extends PlayerBucketEvent {
014    private static final HandlerList handlers = new HandlerList();
015
016    public PlayerBucketFillEvent(final Player who, final Block blockClicked, final BlockFace blockFace, final Material bucket, final ItemStack itemInHand) {
017        super(who, blockClicked, blockFace, bucket, itemInHand);
018    }
019
020    @Override
021    public HandlerList getHandlers() {
022        return handlers;
023    }
024
025    public static HandlerList getHandlerList() {
026        return handlers;
027    }
028}
